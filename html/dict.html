1. 什么是字典

1.1 基本概念

字典是 Python 中一种键值对(key-value)的数据结构。你可以把它想象成一本真实的字典：

  • 键(key) = 你要查找的单词

  • 值(value) = 单词的解释

# 一个简单的字典
student = {
    "name": "小明",
    "age": 18,
    "grade": "高三"
}

1.2 字典的特点

  • 无序性：Python 3.7+ 保持插入顺序，但不应依赖位置

  • 键唯一：每个键只能出现一次

  • 键不可变：键必须是不可变类型（字符串、数字、元组）

  • 值任意：值可以是任何数据类型

  • 可变性：字典本身可以修改

xxxxxxxxxx
# 键必须唯一，重复的键会覆盖
info = {
    "name": "小红",
    "name": "小蓝"  # 这会覆盖上面的值
}
print(info)  # {'name': '小蓝'}

2.2 使用 dict() 函数

xxxxxxxxxx

# 方式2：列表的元组

print(fruits)  # {'apple': 5, 'banana': 3, 'orange': 8}


# 方式3：两个列表组合
keys = ["a", "b", "c"]
values = [1, 2, 3]
my_dict = dict(zip(keys, values))
print(my_dict)  # {'a': 1, 'b': 2, 'c': 3}

2.3 使用 fromkeys() 创建

3. 访问字典元素

3.1 使用方括号 []

xxxxxxxxxx
student = {
    "name": "王五",

}


# 获取值

xxxxxxxxxx
student = {

# get() 方法更安全
print(student.get("name"))      # 王五
print(student.get("grade"))     # None（键不存在返回 None）
print(student.get("grade", "未知"))  # 未知（可以设置默认值）

4. 修改字典

4.1 添加/修改元素

xxxxxxxxxx
student = {"name": "小明", "age": 18}


# 添加新键值对
student["grade"] = "高三"
student["school"] = "第一中学"
print(student)
# {'name': '小明', 'age': 18, 'grade': '高三', 'school': '第一中学'}


# 修改已有键的值
student["age"] = 19
print(student["age"])  # 19

4.2 使用 update() 批量更新

xxxxxxxxxx
student = {"name": "小明", "age": 18}


# 批量添加/更新
student.update({
    "grade": "高三",
    "score": 95,
    "age": 19  # 这会更新已有的 age
})
print(student)
# {'name': '小明', 'age': 19, 'grade': '高三', 'score': 95}


# 也可以用关键字参数
student.update(city="北京", hobby="篮球")
print(student)

4.3 使用 setdefault()

xxxxxxxxxx
# setdefault：如果键不存在就添加，存在就返回原值
student = {"name": "小明"}


# 键不存在，添加并返回新值
age = student.setdefault("age", 18)
print(age)       # 18
print(student)   # {'name': '小明', 'age': 18}


# 键已存在，返回原值，不修改
name = student.setdefault("name", "小红")
print(name)      # 小明（不是小红）
print(student)   # {'name': '小明', 'age': 18}

5. 删除操作

5.1 使用 del 语句

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}


# 删除指定键值对
del student["grade"]
print(student)  # {'name': '小明', 'age': 18}


# 删除整个字典
# del student
# print(student)  # NameError: name 'student' is not defined

5.2 使用 pop() 方法

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}


# pop() 删除并返回值
age = student.pop("age")
print(age)       # 18
print(student)   # {'name': '小明', 'grade': '高三'}


# 键不存在时可以设置默认值
score = student.pop("score", 0)
print(score)     # 0（键不存在，返回默认值）

5.3 使用 popitem() 方法

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}


# popitem() 删除并返回最后一个键值对（元组形式）
last_item = student.popitem()
print(last_item)  # ('grade', '高三')
print(student)    # {'name': '小明', 'age': 18}

5.4 使用 clear() 清空字典

xxxxxxxxxx
student = {"name": "小明", "age": 18}


student.clear()
print(student)  # {}（空字典，但变量还在）

6. 字典常用方法

6.1 keys() - 获取所有键

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}


keys = student.keys()
print(keys)        # dict_keys(['name', 'age', 'grade'])
print(list(keys))  # ['name', 'age', 'grade']

6.2 values() - 获取所有值

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}


values = student.values()
print(values)        # dict_values(['小明', 18, '高三'])
print(list(values))  # ['小明', 18, '高三']

6.3 items() - 获取所有键值对

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}


items = student.items()
print(items)        # dict_items([('name', '小明'), ('age', 18), ('grade', '高三')])
print(list(items))  # [('name', '小明'), ('age', 18), ('grade', '高三')]

6.4 copy() - 浅拷贝

xxxxxxxxxx
original = {"name": "小明", "scores": [90, 85, 88]}


# 浅拷贝
copied = original.copy()
copied["name"] = "小红"
print(original["name"])  # 小明（不受影响）


# 注意：嵌套对象是共享的
copied["scores"].append(95)
print(original["scores"])  # [90, 85, 88, 95]（受影响了！）

6.5 len() - 获取字典长度

xxxxxxxxxx
student = {"name": "小明", "age": 18, "grade": "高三"}
print(len(student))  # 3

8. 字典推导式

字典推导式是创建字典的简洁方式，类似列表推导式。

8.1 基本语法

xxxxxxxxxx
# 语法：{键表达式: 值表达式 for 变量 in 可迭代对象}


# 示例：创建平方字典
squares = {x: x**2 for x in range(1, 6)}
print(squares)  # {1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

8.2 带条件的推导式

xxxxxxxxxx
# 只保留偶数的平方
even_squares = {x: x**2 for x in range(1, 11) if x % 2 == 0}
print(even_squares)  # {2: 4, 4: 16, 6: 36, 8: 64, 10: 100}

8.3 实际应用示例

xxxxxxxxxx
# 示例1：将列表转为字典（值为索引）
fruits = ["apple", "banana", "orange"]
fruit_index = {fruit: index for index, fruit in enumerate(fruits)}
print(fruit_index)  # {'apple': 0, 'banana': 1, 'orange': 2}


# 示例2：反转字典的键和值
original = {"a": 1, "b": 2, "c": 3}
reversed_dict = {value: key for key, value in original.items()}
print(reversed_dict)  # {1: 'a', 2: 'b', 3: 'c'}


# 示例3：过滤字典
scores = {"小明": 85, "小红": 92, "小蓝": 78, "小绿": 95}
passed = {name: score for name, score in scores.items() if score >= 80}
print(passed)  # {'小明': 85, '小红': 92, '小绿': 95}


# 示例4：转换值
prices = {"apple": 5, "banana": 3, "orange": 8}
discounted = {item: price * 0.8 for item, price in prices.items()}
print(discounted)  # {'apple': 4.0, 'banana': 2.4, 'orange': 6.4}

10.5 简单的购物车

xxxxxxxxxx
cart = {}


def add_to_cart(item, quantity, price):
    if item in cart:
        cart[item]["quantity"] += quantity
    else:
        cart[item] = {"quantity": quantity, "price": price}


def get_total():
    total = 0
    for item, info in cart.items():
        total += info["quantity"] * info["price"]
    return total


def show_cart():
    print("购物车内容：")
    for item, info in cart.items():
        subtotal = info["quantity"] * info["price"]
        print(f"  {item}: {info['quantity']}个 x {info['price']}元 = {subtotal}元")
    print(f"总计: {get_total()}元")


# 测试
add_to_cart("苹果", 3, 5)
add_to_cart("香蕉", 2, 3)
add_to_cart("苹果", 2, 5)  # 再加2个苹果
show_cart()

输出：

xxxxxxxxxx

11. 常见错误与注意事项

11.1 KeyError 错误

xxxxxxxxxx
student = {"name": "小明"}


# 错误：访问不存在的键
# print(student["age"])  # KeyError: 'age'


# 正确：使用 get() 或先检查
print(student.get("age", "未知"))  # 未知


if "age" in student:
    print(student["age"])
else:
    print("没有 age 字段")

11.2 可变对象作为键

xxxxxxxxxx
# 错误：列表不能作为键（列表是可变的）
# my_dict = {[1, 2]: "value"}  # TypeError


# 正确：使用元组（元组是不可变的）
my_dict = {(1, 2): "value"}
print(my_dict[(1, 2)])  # value

11.3 遍历时修改字典

xxxxxxxxxx
# 错误：遍历时直接删除
student = {"name": "小明", "age": 18, "grade": "高三"}
# for key in student:
#     if key == "age":
#         del student[key]  # RuntimeError


# 正确：遍历副本或收集要删除的键
student = {"name": "小明", "age": 18, "grade": "高三"}
keys_to_delete = [key for key in student if key == "age"]
for key in keys_to_delete:
    del student[key]
print(student)  # {'name': '小明', 'grade': '高三'}

11.4 浅拷贝的陷阱

xxxxxxxxxx
original = {"info": {"name": "小明", "age": 18}}
copied = original.copy()


# 修改嵌套对象会影响原字典
copied["info"]["age"] = 20
print(original["info"]["age"])  # 20（被修改了！）


# 解决方案：使用深拷贝
import copy
original = {"info": {"name": "小明", "age": 18}}
deep_copied = copy.deepcopy(original)
deep_copied["info"]["age"] = 20
print(original["info"]["age"])  # 18（不受影响）
